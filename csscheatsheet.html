Selectors ▼

/* Element Selector */ h1 { color: blue; } /* Class Selector */ .className { color: red; } /* ID Selector */ #idName { color: green; } /* Universal Selector */ * { margin: 0; } /* Attribute Selector */ [type="text"] { border: 1px solid; } /* Descendant Selector */ div p { color: blue; } /* Child Selector */ div > p { color: red; } /* Adjacent Sibling */ h1 + p { margin-top: 0; } /* General Sibling */ h1 ~ p { color: gray; } /* Pseudo-classes */ a:hover { color: orange; } input:focus { border-color: blue; } li:first-child { font-weight: bold; } li:last-child { border: none; } li:nth-child(2n) { background: gray; }

Flexbox ▼

/* Container Properties */ .container { display: flex; flex-direction: row; /* row, column, row-reverse, column-reverse */ justify-content: center; /* flex-start, flex-end, center, space-between, space-around */ align-items: center; /* flex-start, flex-end, center, stretch, baseline */ flex-wrap: wrap; /* nowrap, wrap, wrap-reverse */ gap: 10px; } /* Item Properties */ .item { flex: 1; /* flex-grow flex-shrink flex-basis */ flex-grow: 1; flex-shrink: 1; flex-basis: auto; align-self: flex-start; order: 1; }

Responsive Design ▼

/* Media Queries */ @media (max-width: 768px) { .container { flex-direction: column; } } @media (min-width: 769px) and (max-width: 1024px) { .sidebar { width: 200px; } } /* Common Breakpoints */ /* Mobile: max-width: 480px */ /* Tablet: 481px - 768px */ /* Desktop: 769px+ */ /* Viewport Units */ width: 100vw; /* viewport width */ height: 100vh; /* viewport height */ font-size: 5vmin; /* 5% of smaller dimension */

CSS Variables ▼

/* Define Variables */ :root { --primary-color: #007bff; --secondary-color: #6c757d; --font-size-base: 16px; --spacing-unit: 8px; } /* Use Variables */ .button { background-color: var(--primary-color); font-size: var(--font-size-base); padding: calc(var(--spacing-unit) * 2); } /* Fallback Values */ color: var(--text-color, black);
